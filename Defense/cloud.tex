\begin{slide}{UAV Chemical Cloud Tracking}
  UAV Characteristics
  \begin{itemize}
  \item UAV flight model
    \begin{itemize}
    \item Point mass
    \item
      $\left\{ 
      \begin{array}{l}
        \dot {x}=v_C \cos (\theta ) \\ 
        \dot {y}=v_C \sin (\theta ) \\ 
        \dot {\theta}=\omega \\
      \end{array}
      \right.$
    \item Fixed turning radius
    \item Constant speed
    \item Geometric path planner
    \end{itemize}
  \item 30 minute power supply
  \item Binary chemical sensor
  \end{itemize}
\end{slide}
